API Documentation

JavaScript API reference for SIRVATO OS components

Theme API

theme.js
// Toggle theme programmatically
document.body.classList.toggle('light-mode');

// Read current theme
const isDark = !document.body.classList.contains('light-mode');

// Set theme explicitly
document.body.classList.remove('light-mode');   // dark
document.body.classList.add('light-mode');       // light

// Theme is persisted automatically via localStorage key: 'theme'

i18n API

i18n.js
// Mark an element for translation using data-i18n attribute
// <span data-i18n="nav.frameworks">Frameworks</span>

// Language is persisted via localStorage key: 'lang'
// Supported values: 'en', 'it'

// Translations object structure:
const translations = {
  en: { 'nav.frameworks': 'Frameworks', ... },
  it: { 'nav.frameworks': 'Framework', ... }
};

// Toggle language programmatically
document.getElementById('lang-toggle').click();

Demo API

demo.js — generatePrompt()
// The demo page exposes a global generatePrompt() function
// It reads the 5 MICRO input fields and builds a structured prompt

generatePrompt();
// Returns: string — the fully assembled MICRO prompt

// Field IDs used:
// #input-message   → M — Message
// #input-intention → I² — Intention
// #input-context   → C — Context
// #input-rhythm    → R — Rhythm
// #input-output    → O — Output

// Output is written to: #prompt-output
// Copy button ID: #copy-prompt

Animations API

animations.js
// Elements with class 'reveal' animate on scroll (fade-in + slide-up)
// Elements with class 'reveal-group' animate children in sequence

// Add reveal to any element:
// <div class="reveal">...</div>

// Start matrix rain on a canvas element:
// <canvas id="matrix-rain"></canvas>
// The canvas is detected automatically on pages that include it

// Pulse animation: add class 'pulse-neon' to any element
// <span class="pulse-neon">LIVE</span>

Copy-to-Clipboard API

main.js — Copy Buttons
// All buttons with class 'code-copy' are wired automatically by main.js
// They copy the text content of the nearest <pre><code> sibling

// Markup pattern:
// <div class="code-block">
//   <div class="code-header">
//     <span class="code-label">Label</span>
//     <button class="code-copy">Copy</button>
//   </div>
//   <pre><code>...content...</code></pre>
// </div>

// Button shows "Copied!" for 2 seconds after a successful copy
